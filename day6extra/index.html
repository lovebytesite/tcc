TIC-80⌗ 

function TIC()
 cls()
 t=time()/200
 sin=math.sin
 
 for ly=1,8 do
  sv=sin(t/2+ly/4)
  for lx=1,8 do
   line(0,68+lx+sv*50,240,68+lx+sv*50,lx+8)
  end
 end
 
 for i=1,67 do
  sv=sin(i/13+t/8)*sin(i/7+t/2)*60
  for j=0,8 do
   line(sv+j+120,0+i*2,sv+j+120,136,j+8)
  end
 end
end
